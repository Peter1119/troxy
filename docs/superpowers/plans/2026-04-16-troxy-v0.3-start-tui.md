```python
"""Parse filter expressions like 'host:X status:4xx method:POST'."""

import re


def parse_filter(text: str) -> dict:
    if not text or not text.strip():
        return {}

    result = {}
    freetext_parts = []
    tokens = text.strip().split()

    for token in tokens:
        if ":" in token:
            key, value = token.split(":", 1)
            key = key.lower()

            if key == "host":
                result["domain"] = value
            elif key == "status":
                match = re.match(r"^(\d)xx$", value, re.IGNORECASE)
                if match:
                    base = int(match.group(1)) * 100
                    result["status_range"] = (base, base + 99)
                else:
                    result["status"] = int(value)
            elif key == "method":
                result["method"] = value.upper()
            elif key == "path":
                result["path"] = value
            else:
                freetext_parts.append(token)
        else:
            freetext_parts.append(token)

    if freetext_parts:
        result["query"] = " ".join(freetext_parts)

    return result
```

```python
from troxy.core.filter_parser import parse_filter


def list_flows_filtered(db_path: str, filter_text: str, *, limit: int = 500) -> list[dict]:
    parsed = parse_filter(filter_text)
    if not parsed:
        return list_flows(db_path, limit=limit)

    conn = get_connection(db_path)
    conditions = []
    params = []

    if "domain" in parsed:
        conditions.append("host LIKE ?")
        params.append(f"%{parsed['domain']}%")
    if "status" in parsed:
        conditions.append("status_code = ?")
        params.append(parsed["status"])
    if "status_range" in parsed:
        lo, hi = parsed["status_range"]
        conditions.append("status_code BETWEEN ? AND ?")
        params.extend([lo, hi])
    if "method" in parsed:
        conditions.append("UPPER(method) = ?")
        params.append(parsed["method"])
    if "path" in parsed:
        conditions.append("path LIKE ?")
        params.append(parsed["path"].replace("*", "%"))

    where = " AND ".join(conditions) if conditions else "1=1"
    sql = f"SELECT * FROM flows WHERE {where} ORDER BY timestamp DESC LIMIT ?"
    params.append(limit)

    rows = conn.execute(sql, params).fetchall()
    conn.close()

    results = [dict(r) for r in rows]

    if "query" in parsed:
        q = parsed["query"].lower()
        results = [
            r for r in results
            if q in (r.get("request_body") or "").lower()
            or q in (r.get("response_body") or "").lower()
            or q in (r.get("path") or "").lower()
        ]

    return results
```

```python
import re

_DYNAMIC_SEGMENT = re.compile(
    r"^("
    r"\d+$"           # pure numeric: 123, 2026
    r"|"
    r"[0-9a-f]{8}-"   # UUID prefix
    r"|"
    r"[A-Za-z0-9]{10,}" # long alphanumeric (base64-ish IDs)
    r")"
)


def suggest_glob(path: str) -> str:
    segments = path.strip("/").split("/")
    result = []
    for seg in segments:
        if _DYNAMIC_SEGMENT.match(seg):
            result.append("*")
        else:
            result.append(seg)
    return "/" + "/".join(result)
```

```python
"""Manage mitmdump subprocess lifecycle."""

import os
import shutil
import signal
import subprocess
import sys


class ProxyManager:
    def __init__(self, port: int = 8080, mode: str | None = None):
        self._port = port
        self._mode = mode
        self._process: subprocess.Popen | None = None

    def _find_mitmdump(self) -> str | None:
        venv_bin = os.path.join(os.path.dirname(sys.executable), "mitmdump")
        if os.path.exists(venv_bin):
            return venv_bin
        return shutil.which("mitmdump")

    def _addon_path(self) -> str:
        return os.path.join(
            os.path.dirname(os.path.dirname(os.path.abspath(__file__))),
            "addon.py",
        )

    def start(self) -> None:
        mitmdump = self._find_mitmdump()
        if not mitmdump:
            raise RuntimeError("mitmdump not found. Install: uv add mitmproxy")

        cmd = [mitmdump, "-s", self._addon_path(), "-p", str(self._port), "-q"]
        if self._mode:
            cmd.extend(["--mode", self._mode])

        self._process = subprocess.Popen(
            cmd,
            stdout=subprocess.DEVNULL,
            stderr=subprocess.PIPE,
        )

    def stop(self) -> None:
        if self._process and self._process.poll() is None:
            self._process.send_signal(signal.SIGTERM)
            try:
                self._process.wait(timeout=5)
            except subprocess.TimeoutExpired:
                self._process.kill()
            self._process = None

    @property
    def running(self) -> bool:
        return self._process is not None and self._process.poll() is None
```

```python
        super().__init__()
        self._db_path = db_path
        self._flow_id = flow_id
        self._flow: dict | None = None

    def compose(self) -> ComposeResult:
        yield Static(id="url-bar")
        yield Vertical(
            Static(id="request-pane", classes="pane"),
            id="request-container",
        )
        yield Vertical(
            Static(id="response-pane", classes="pane"),
            id="response-container",
        )
        yield Static(copy.DETAIL_HINT, id="hint-bar")

    DEFAULT_CSS = """
    #request-container {
        height: 1fr;
        border-top: solid $primary;
    }
    #response-container {
        height: 2fr;
        border-top: solid $primary;
    }
    .pane {
        overflow-y: scroll;
    }
    .focused-pane {
        border-left: thick $accent;
    }
    """

    def on_mount(self) -> None:
        self._flow = get_flow(self._db_path, self._flow_id)
        if not self._flow:
            self.app.pop_screen()
            return
        self._render_url()
        self._render_request()
        self._render_response()
        self._update_focus_style()

    def _render_url(self) -> None:
        f = self._flow
        url = f"{f['scheme']}://{f['host']}"
        if f.get("port") and f["port"] not in (80, 443):
            url += f":{f['port']}"
        url += f["path"]
        if f.get("query"):
            url += f"?{f['query']}"
        title = f"Flow #{f['id']} · {f['method']} · {f['status_code']} · {f['duration_ms']:.0f}ms"
        bar = self.query_one("#url-bar", Static)
        bar.update(f"{title}\nURL: {url}")

    def _render_request(self) -> None:
        f = self._flow
        lines = [f"── Request ▌──"]
        headers = json.loads(f["request_headers"]) if isinstance(f["request_headers"], str) else f["request_headers"]
        for k, v in headers.items():
            lines.append(f"{k}: {v}")
        if f.get("request_body"):
            lines.append("")
            lines.append(self._format_body(f["request_body"], f.get("request_content_type")))
        else:
            lines.append("\n(body 없음)")
        self.query_one("#request-pane", Static).update("\n".join(lines))

    def _render_response(self) -> None:
        f = self._flow
        lines = [f"── Response ({f['status_code']}) · {self._format_size(f.get('response_body'))} ──"]
        headers = json.loads(f["response_headers"]) if isinstance(f["response_headers"], str) else f["response_headers"]
        for k, v in headers.items():
            lines.append(f"{k}: {v}")
        if f.get("response_body"):
            lines.append("")
            lines.append(self._format_body(f["response_body"], f.get("response_content_type")))
        self.query_one("#response-pane", Static).update("\n".join(lines))

    def _format_body(self, body: str, content_type: str | None) -> str:
        if body and body.startswith("b64:"):
            return f"(binary, {len(body)} bytes base64)"
        if content_type and "json" in content_type:
            try:
                return json.dumps(json.loads(body), indent=2, ensure_ascii=False)
            except (json.JSONDecodeError, TypeError):
                pass
        return body or ""

    def _format_size(self, body: str | None) -> str:
        if not body:
            return "0 bytes"
        size = len(body.encode("utf-8", errors="replace"))
        if size < 1024:
            return f"{size} bytes"
        return f"{size / 1024:.1f} KB"

    def action_switch_focus(self) -> None:
        self._focused_pane = "request" if self._focused_pane == "response" else "response"
        self._update_focus_style()

        res = self.query_one("#response-pane", Static)
        req.set_class(self._focused_pane == "request", "focused-pane")
        res.set_class(self._focused_pane == "response", "focused-pane")

    def action_go_back(self) -> None:
        self.app.pop_screen()

            url += f":{f['port']}"
```

```python
def _build_request_text(self) -> str:
    f = self._flow
    lines = [f"{f['method']} {self._get_url()}"]
    headers = json.loads(f["request_headers"]) if isinstance(f["request_headers"], str) else f["request_headers"]
    for k, v in headers.items():
        lines.append(f"{k}: {v}")
    if f.get("request_body"):
        lines.append("")
        lines.append(f["request_body"])
    return "\n".join(lines)

    f = self._flow
    lines = [f"HTTP {f['status_code']}"]
    headers = json.loads(f["response_headers"]) if isinstance(f["response_headers"], str) else f["response_headers"]
    for k, v in headers.items():
        lines.append(f"{k}: {v}")
    if f.get("response_body"):
        lines.append("")
        lines.append(f["response_body"])
    return "\n".join(lines)
```

```python
"""MockDialog — flow를 mock rule로 등록하는 다이얼로그."""

import json
from textual.app import ComposeResult
from textual.message import Message
from textual.screen import ModalScreen
from textual.widgets import Input, Static, TextArea, Select, Switch
from textual.containers import Vertical, Horizontal

from troxy.core.mock import suggest_glob, add_mock_rule
from troxy.tui import copy


STATUS_OPTIONS = [
    (str(s), str(s)) for s in [200, 201, 204, 400, 401, 403, 404, 500, 502, 503]

class MockDialog(ModalScreen):
    class Saved(Message):
        def __init__(self, rule_id: int, name: str) -> None:
            self.rule_id = rule_id
            self.name = name
            super().__init__()

    BINDINGS = [
        ("escape", "cancel", "cancel"),
    ]

    DEFAULT_CSS = """
    MockDialog {
        align: center middle;
    }
    #mock-dialog-container {
        width: 70;
        height: auto;
        max-height: 30;
        background: $surface;
        border: solid $primary;
        padding: 1 2;
    }
    """

    def __init__(self, db_path: str, flow: dict, rule: dict | None = None) -> None:
        super().__init__()
        self._db_path = db_path
        self._flow = flow
        self._rule = rule
        self._use_glob = True

    def compose(self) -> ComposeResult:
        f = self._flow
        suggested_glob = suggest_glob(f["path"])
        has_glob = suggested_glob != f["path"]
        default_name = self._auto_name(f)

        if self._rule:
            initial_name = self._rule.get("name") or default_name
            initial_status = str(self._rule.get("status_code", f["status_code"]))
            initial_body = self._rule.get("response_body") or f.get("response_body") or ""
        else:
            initial_name = default_name
            initial_status = str(f["status_code"])
            initial_body = f.get("response_body") or ""

        with Vertical(id="mock-dialog-container"):
            yield Static(f"Mock from Flow #{f['id']}" if not self._rule else f"Edit Mock #{self._rule['id']}")
            yield Static("")
            yield Static("Name:")
            yield Input(value=initial_name, id="mock-name")
            yield Static("")
            yield Static(f"Match: {f['method']}  {f['host']}  {suggested_glob if has_glob else f['path']}")
            if has_glob:
                yield Horizontal(
                    Switch(value=True, id="use-glob"),
                    Static(f" use glob: {suggested_glob}"),
                )
            yield Static("")
            yield Static("Status:")
            yield Select(
                [(s, s) for s in ["200", "201", "204", "400", "401", "403", "404", "500", "502", "503"]],
                value=initial_status,
                id="mock-status",
            )
            yield Static("")
            yield Static("Body:")
            yield TextArea(initial_body, id="mock-body", language="json")
            yield Static("")
            yield Static("Tab next field · ⏎ save · Esc cancel")

    def _auto_name(self, flow: dict) -> str:
        path_parts = flow["path"].strip("/").split("/")
        last_segment = path_parts[-1] if path_parts else "mock"
        if last_segment.isdigit() or len(last_segment) > 10:
            last_segment = path_parts[-2] if len(path_parts) > 1 else "mock"
        return f"{last_segment}-{flow['status_code']}"

    def key_enter(self) -> None:
        self._save()

    def _save(self) -> None:
        f = self._flow
        name = self.query_one("#mock-name", Input).value.strip() or None
        status_str = self.query_one("#mock-status", Select).value
        status_code = int(status_str) if status_str else f["status_code"]
        body = self.query_one("#mock-body", TextArea).text

        use_glob_switch = self.query("#use-glob")
        if use_glob_switch:
            self._use_glob = use_glob_switch.first().value

        path_pattern = suggest_glob(f["path"]) if self._use_glob else f["path"]

        headers = f.get("response_headers")
        if isinstance(headers, str):
            headers = headers
        else:
            headers = json.dumps(headers) if headers else None

        rule_id = add_mock_rule(
            self._db_path,
            domain=f["host"],
            path_pattern=path_pattern,
            method=f["method"],
            status_code=status_code,
            response_headers=headers,
            response_body=body,
            name=name,
        )

        self.post_message(self.Saved(rule_id, name or f"rule-{rule_id}"))
        self.app.pop_screen()

    def action_cancel(self) -> None:
        self.app.pop_screen()
```

```python
        table.clear()
        rules = list_mock_rules(self._db_path)
        for rule in rules:
            icon = MOCK_ENABLED_ICON if rule["enabled"] else MOCK_DISABLED_ICON
            match_str = f"{rule.get('method') or '*'} {rule.get('domain') or '*'}/{rule.get('path_pattern') or '*'}"

                match_str = match_str[:34] + "…"
            table.add_row(
                icon,
                str(rule["id"]),
                rule.get("name") or f"rule-{rule['id']}",
                match_str,
                str(rule["status_code"]),
                str(rule.get("hit_count", 0)),
                key=str(rule["id"]),
            )

    def _get_selected_rule_id(self) -> int | None:
        table = self.query_one("#mock-table", DataTable)
        if table.cursor_row is not None:
            row = table.get_row_at(table.cursor_row)
            return int(row[1]) if row else None
        return None
```
